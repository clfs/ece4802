\section*{Problem 3}
\subsection*{3b}
\begin{itemize}
	\item Gen\((kw, pt)\) outputs the keyword and plaintext, concatenated and truncated.
	\item Enc\((kw, pt) = (pt_i + k_i)\pmod{26}\) for each \(i\).
	\item Dec\((kw, ct) = (ct_i - k_i)\pmod{26}\) for each \(i\).
\end{itemize}
